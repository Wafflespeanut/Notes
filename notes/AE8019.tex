\para{\LARGE Multi-stage rockets:}
{\LARGE
$$\mathrm{Mass\ ratio,\ \Lambda}_i = \frac{M_{0i}}{M_{0i} - M_{i}} $$
$$\mathrm{Payload\ ratio,\ \lambda}_i = \frac{M_{0i + 1}}{M_{0i}} $$
$$\mathrm{Propellant\ ratio,\ \psi}_i = \frac{M_{pi}}{M_{0i}} $$
$$\mathrm{Structural\ efficiency,\ \epsilon}_i = \frac{M_{ci}}{M_{ci} + M_{pi}} $$ \
$$\psi_i = (1 - \epsilon_i)(1 - \lambda_i),\ \ \psi_{total} = \frac{\sum M_{pi}}{M_{0i}}$$ \
$$\Lambda_i = \frac{1}{1 - \psi_i} = \frac{1}{\epsilon_i(1 - \lambda_i) + \lambda_i}$$ \
$$\lambda_{total} = \frac{M_{uN}}{M_{0i}},\ \ G = \frac{M_0}{M_u}$$
}
\paragraph{\LARGE Basic stuff:}
{\LARGE $$ F = \dot{m} g\ I_{sp} $$
$$ \Lambda = \frac{M_0}{M_0 - \dot{m}t} $$}
(for constant thrust),
{\LARGE $$ P(t) = 1 - \frac{1}{\Lambda} \big(\mathrm{ln} \Lambda + 1\big)$$
$$ \psi_0 = \frac{F}{M_0g},\ t_b = \frac{I_{sp}}{\psi_0}\ \bigg(1 - \frac{1}{\Lambda} \bigg)$$} \\
(for constant specific thrust),
{\LARGE $$ \beta_0 = \frac{F}{Mg},\ \dot{m} = \frac{M_0\beta_0}{I_{sp}\Lambda},\ t_b = \frac{I_{sp}}{\beta_0}\ \mathrm{ln} \Lambda $$}
\section{\LARGE Vertical flight:}
{\LARGE $$ v = g\ I_{sp}\ \mathrm{ln} \Lambda - gt,\ \ t_c = I_{sp}\  \mathrm{ln} \Lambda$$
$$ h(t) = g\ \frac{I_{sp} ^ 2}{\psi_0}\ P(t) - \frac{1}{2}\ g t^2 $$
$$ h(t_c) = g\ \frac{I_{sp} ^ 2}{\psi_0}\ \bigg(\frac{1}{2}\ \psi_0\ (\mathrm{ln} \Lambda) ^ 2 - \mathrm{ln} \Lambda + 1 - \frac{1}{\Lambda} \bigg) $$
$$ h = h_b + v_b (t - t_b) - \frac{1}{2}\ g\ (t - t_b)^2 $$
$$ (h_b)_{max} = g\ \frac{I_{sp} ^ 2}{2} \frac{P_b^2}{(1 - \frac{1}\Lambda) ^ 2} $$
}
\section{\LARGE Constant pitch angle:}
{\LARGE
$$ v_x = g\ I_{sp}\ \mathrm{ln} \Lambda\ cos \theta,\ v_z = g\ I_{sp}\ \mathrm{ln} \Lambda\ sin \theta - g t$$
$$ v_b = \Delta v_{ideal} \sqrt{1 - \frac{2t_b\ sin\theta}{I_{sp}\ \mathrm{ln}\Lambda} - \frac{t_b ^ 2}{I_{sp}^2(\mathrm{ln}\Lambda)^2}} $$
$$ tan \alpha = \frac{t\ cos \theta}{I_{sp}\ \mathrm{ln}\Lambda - t\ sin\theta} $$}
(for constant thrust),
{\LARGE $$ tan \alpha = \frac{cos \theta}{ \psi_0 - sin\theta} $$}
(for constant specific thrust),
{\LARGE $$ tan \alpha = \frac{cos \theta}{ \beta_0 - sin\theta} $$
$$ M = M_0\ \mathrm{exp}\bigg( \frac{-\beta_0}{I_{sp}}\ t \bigg)$$}
\para{\Large (relation to vertical flight)}
{\LARGE $$ t_c' = t_c\ sin\theta,\ x(t) = h\ cos\theta\ (\mathrm{no\ g!}),\ z(t) = h\ sin\theta$$}
\centerline{(same relation goes for $v_x$ \& $v_z$ and $x_c$ \& $z_c$)} \\

\section{\LARGE Gravity turn:}
{\LARGE $$ \frac{dv}{dt} = \frac{F}{m} - g\ sin\gamma = g(\beta_0 - sin\gamma)$$ \
$$ \frac{d\gamma}{dt} = \frac{-g}{v} cos\gamma$$ \
$$ \Gamma = \frac{1}{cos\gamma} \bigg( \frac{1 - sin\gamma}{1 + sin\gamma} \bigg) ^ {\beta_0 / 2} $$ \
$$ \frac{dv}{dt} = \bigg(\frac{\beta_0 - \sin \gamma}{cos \gamma}\bigg)\ d\gamma $$ \
$$ t = \frac{v\ (\beta_0 + sin\gamma)}{g(\beta_0 ^ 2 - 1)} $$
}
